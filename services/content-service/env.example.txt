# Content Service Environment Variables (example for local development)

# Application
NODE_ENV=development
PORT=8083
HOST=0.0.0.0
LOG_LEVEL=debug

# Example: postgresql://username:password@localhost:5432/marketplace_db

# Redis (optional)
REDIS_HOST=localhost

# Kafka (optional)
# Comma separated list of brokers
KAFKA_BROKERS=localhost:9092
KAFKA_CLIENT_ID=content-service
KAFKA_GROUP_ID=content-service-group

# JWT (shared with Marketplace Service)
JWT_SECRET=your-super-secret-jwt-key-change-in-production
JWT_ISSUER=marketplace-service

# Marketplace Service Integration (used by some features)
MARKETPLACE_SERVICE_URL=http://localhost:8080/api

# File Storage (local or s3)
# STORAGE_TYPE can be 'local' or 's3'
STORAGE_TYPE=local
STORAGE_LOCAL_PATH=./uploads

# Include your frontend development origin(s). In development, http://localhost:3002 is added by default.
